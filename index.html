nums = input("Enter numbers: ").split(',')
nums = [int(x) for x in nums]
n = len(nums)

mean = sum(nums) / n

nums.sort()
median = (nums[n//2] + nums[(n-1)//2]) / 2

freq = {}
for x in nums:
    freq[x] = freq.get(x, 0) + 1

maxf = max(freq.values())
mode = [k for k, v in freq.items() if v == maxf]
mode = mode if maxf > 1 else "No mode"

var = sum((x - mean) ** 2 for x in nums) / n
std = var ** 0.5

print("Mean =", mean)
print("Median =", median)
print("Mode =", mode)
print("Variance =", var)
print("Std Dev =", std)
