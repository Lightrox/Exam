from pywinauto import Application, Desktop
from time import sleep
Application(backend="uia").start("calc.exe")
sleep(3)
calc = Desktop(backend="uia").window(title_re=".*Calculator.*")
calc.wait("visible", timeout=15)
for b in ["Seven", "Plus", "Three", "Equals"]:
    calc.child_window(title=b, control_type="Button").click()
sleep(1)
result = calc.child_window(auto_id="CalculatorResults", control_type="Text")\
             .window_text().replace("Display is", "").strip()
print("Test Passed" if result == "10" else f"Test Failed: {result}")
calc.close()
